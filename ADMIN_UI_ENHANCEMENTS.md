## API Integration Examples

### React Hook for Voice Config
```typescript
import { useQuery, useMutation } from '@tanstack/react-query';

const VOICE_API = 'http://192.168.0.151:7101';

export function useVoiceConfig() {
  const { data: config } = useQuery({
    queryKey: ['voice-config'],
    queryFn: () => fetch(`${VOICE_API}/api/config`).then(r => r.json()),
  });

  const { data: voices } = useQuery({
    queryKey: ['voices'],
    queryFn: () => fetch(`${VOICE_API}/api/voices`).then(r => r.json()),
  });

  const { data: emotions } = useQuery({
    queryKey: ['emotions'],
    queryFn: () => fetch(`${VOICE_API}/api/emotions`).then(r => r.json()),
  });

  const updateConfig = useMutation({
    mutationFn: (newConfig) => 
      fetch(`${VOICE_API}/api/config`, {
        method: 'PUT',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(newConfig),
      }),
  });

  return { config, voices, emotions, updateConfig };
}
```

### TTS with Emotion in Chat
```typescript
async function speakResponse(text: string, context: 'greeting' | 'error' | 'success') {
  const emotionMap = {
    greeting: 'friendly',
    error: 'empathetic',
    success: 'happy',
  };

  const response = await fetch(`${VOICE_API}/api/speak`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      text,
      language: 'hi',
      emotion: emotionMap[context],
      speed: 1.0,
    }),
  });

  const audioBlob = await response.blob();
  const audio = new Audio(URL.createObjectURL(audioBlob));
  audio.play();
}
```
